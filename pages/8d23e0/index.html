21.【简单】合并两个有序链表

文章发布较早，内容可能过时，阅读注意甄别。

# 题目

将两个升序链表合并为一个新的 升序 链表并返回。新链表是通过拼接给定的两个链表的所有节点组成的。

示例：

输入：1->2->4, 1->3->4
输出：1->1->2->3->4->4
1
2

# 分析

此题较为简单，首先是要判断链表空的情况 然后注意记录head头防止丢失

# 题解

/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func mergeTwoLists(l1 *ListNode, l2 *ListNode) *ListNode {
    if l1 == nil {
        return l2
    } 
    if l2 == nil {
        return l1
    }

    var head, node *ListNode

    if l1.Val < l2.Val {
        head = l1
        node = l1
        l1 = l1.Next
    } else {
        head = l2
        node = l2
        l2 = l2.Next
    }

    for l1 != nil && l2 != nil {
        if l1.Val < l2.Val {
            node.Next = l1
            l1 = l1.Next
        } else {
            node.Next = l2
            l2 = l2.Next
        }
       
        node = node.Next
    }

    if l1 !=nil{
        node.Next = l1
    }
    if l2 != nil {
         node.Next = l2
    }
    return head
}
